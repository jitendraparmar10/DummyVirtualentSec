// --- Helper for random values ---
const getRandomValue = (min, max) => Math.random() * (max - min) + min;

// --- Background Text for Breadcrumb ---
const backgroundTexts = ['Innovation', 'Code', 'Design', '2023', 'Creative', 'React', 'Modern', 'Future', 'Develop'];

const App = () => {
  const textRef = useRef(null);

  useEffect(() => {
    const handleScroll = () => {
      const textElement = textRef.current;
      if (textElement) {
        const { top, height } = textElement.getBoundingClientRect();
        const scrollPercent = Math.max(0, Math.min(1, 1 - (top + height / 1.5) / window.innerHeight));

        const text = textElement.dataset.text;
        const textLength = text.length;
        const lettersToShow = Math.ceil(textLength * scrollPercent);

        let newHtml = '';
        for (let i = 0; i < textLength; i++) {
          if (i < lettersToShow) {
            newHtml += `<span class="black">${text[i]}</span>`;
          } else {
            newHtml += `<span class="grey">${text[i]}</span>`;
          }
        }
        textElement.innerHTML = newHtml;
      }
    };

    window.addEventListener('scroll', handleScroll);
    handleScroll(); // Initial check

    return () => {
      window.removeEventListener('scroll', handleScroll);
    };
  }, []);

  return (
    <div className="App">
      <header className="breadcrumb">
        <div className="breadcrumb-content">
          <h1>About Us</h1>
          <p>Established in 2023</p>
        </div>
        <div className="breadcrumb-bg-particles">
          {backgroundTexts.map((text, index) => (
            <span
              key={index}
              style={{
                top: `${getRandomValue(5, 95)}%`,
                left: `${getRandomValue(5, 95)}%`,
                transform: `rotate(${getRandomValue(-45, 45)}deg)`,
                fontSize: `${getRandomValue(1, 4)}rem`,
              }}
            >
              {text}
            </span>
          ))}
        </div>
      </header>

      <main className="content">
        <section className="about-text-section">
          <p
            ref={textRef}
            data-text="Welcome to our company, a place where innovation and creativity converge. We are a team of passionate developers, designers, and strategists dedicated to crafting exceptional digital experiences. Our journey began with a simple idea: to build beautiful and functional applications that make a difference. We believe in the power of technology to solve complex problems and drive progress. Our approach is collaborative and client-focused, ensuring that we not only meet but exceed expectations."
          >
            {/* Initial text will be rendered grey by CSS */}
            Welcome to our company, a place where innovation and creativity converge. We are a team of passionate developers, designers, and strategists dedicated to crafting exceptional digital experiences. Our journey began with a simple idea: to build beautiful and functional applications that make a difference. We believe in the power of technology to solve complex problems and drive progress. Our approach is collaborative and client-focused, ensuring that we not only meet but exceed expectations.
          </p>
        </section>

        <section className="marquee-section">
          <h2>We Work With</h2>
          <div className="marquee">
            <div className="marquee-content">
              {[...companies, ...companies].map((company, index) => (
                <div className="logo-item" key={index}>
                  <img src={company.logo} alt={company.name} />
                </div>
              ))}
            </div>
          </div>
        </section>

        <section className="mission-vision-section">
          <div className="card mission-card">
            <div className="card-icon">ðŸŽ¯</div>
            <h3>Our Mission</h3>
            <p>
              To empower businesses with innovative and cutting-edge digital
              solutions that drive growth, efficiency, and success. We are
              committed to delivering high-quality products and services that
              exceed client expectations.
            </p>
          </div>
          <div className="card vision-card">
            <div className="card-icon">ðŸ”­</div>
            <h3>Our Vision</h3>
            <p>
              To be a globally recognized leader in the technology industry,
              renowned for our creativity, expertise, and commitment to
              excellence. We aspire to create a future where technology
              enriches lives.
            </p>
          </div>
        </section>

        <section className="marquee-section tech-stack">
          <h2>Our Technology Stack</h2>
          <div className="marquee">
            <div className="marquee-content">
              {[...technologies, ...technologies].map((tech, index) => (
                <div className="tech-item" key={index}>
                  <img src={tech.icon} alt={tech.name} />
                  <span>{tech.name}</span>
                </div>
              ))}
            </div>
          </div>
        </section>
      </main>
    </div>
  );
};

export default App;
